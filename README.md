## Install

```bash
npm i @hardiknaik/react-number-scroll
```

## Usage

```tsx
import React, { Component } from 'react'

import { NumberScroll } from "@hardiknaik/react-number-scroll";

class Example extends Component {
  render() {
    return <NumberScroll number={444} />
  }
}
```
